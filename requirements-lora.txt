# Dedicated environment for the LoRA experiment (lora_data_prep.py + lora_train.py).
# Deliberately does NOT include vllm or sentence-transformers: serving and benchmarking stay
# in `metasra`, this env only builds SFT data and trains adapters. Keeping them apart is the
# point -- transformers 5.x breaks sentence-transformers 5.0, which main.py imports at module
# level (main.py:10), so a shared env takes down the whole evaluation pipeline.
#
#   conda create -n taxolora python=3.12 -y
#   conda activate taxolora
#   pip install -r requirements-lora.txt
#   python verify_lora_env.py          # checks every API the trainer touches
#
# Then freeze what actually resolved, so this is reproducible next time:
#   pip freeze > requirements-lora.lock.txt

# --- Core training stack ---
# transformers >=5.14.1 is a HARD floor, not a preference: google/gemma-4-31b-it is
# model_type "gemma4" and ships extra_special_tokens as a list (['<|video|>']). Anything
# older raises AttributeError: 'list' object has no attribute 'keys' in the tokenizer.
transformers>=5.14.1
torch>=2.11.0

# Left unpinned on purpose. The versions in `metasra` (peft 0.19.0, bitsandbytes 0.49.2,
# accelerate 1.13.0) predate transformers 5.x, and I don't know which specific releases
# first supported it -- guessing an exact pin here would be inventing a number. Let pip
# resolve the latest, confirm with verify_lora_env.py, then lock the result.
accelerate
peft
bitsandbytes

# --- Tokenizer / model IO ---
huggingface-hub>=1.24.0
safetensors>=0.8.0
sentencepiece
protobuf

# --- lora_data_prep.py ---
# networkx alone covers it. Data prep runs fully standalone here: it renders prompts with a
# built-in byte-for-byte copy of our_method's 'full' prompt when our_method can't import, so
# the ontology stack (nltk/obonet/requests/...) is NOT needed just to build SFT data. If
# our_method IS importable it's used directly; test_lora_prep keeps the copy in sync.
# (The built-in copy only covers the 'full' variant -- any other --variant needs the full env.)
networkx>=3.0
